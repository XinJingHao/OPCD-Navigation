360  void _nnSearchRecursive(const PointT& query, const KDNode* node, int* guess, double* min_dist) const
361  {
362  if (node == nullptr)
363  return;
364 
365  const PointT& train = points_[node->idx];
366 
367  const double dist = _distance(query, train);
368  if (dist < *min_dist)
369  {
370  *min_dist = dist;
371  *guess = node->idx;
372  }
373 
374  const int axis = node->axis;
375  const int dir = query[axis] < train[axis] ? 0 : 1;
376  _nnSearchRecursive(query, node->next[dir], guess, min_dist);
377 
378  // if the min distance crosses the axis, the nearest neighbor maybe exist
379  // in the other side of axis, therefore another direction should be searched
380  const double diff = fabs(query[axis] - train[axis]);
381  if (diff < *min_dist)
382  _nnSearchRecursive(query, node->next[!dir], guess, min_dist);
383  }
